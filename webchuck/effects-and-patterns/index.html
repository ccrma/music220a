f => dac.chan(1);

// set filter Q
5 => f.Q;
while( true )
{
    // sweep the cutoff
    1200 + Std.fabs(Math.sin(4.0 * (now/second))) * 400 => f.freq;
    // advance time
    1::samp => now;
}
