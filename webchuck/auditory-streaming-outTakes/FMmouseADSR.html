Homework 2 is about synthesizing an auditory streaming illusion using FM tones.

FM demo

This demo is a real-time FM synth with mouse / touchpad interface.
Move the mouse in X and Y to change modulator frequency and index, respectively.
Click the mouse to mute / unmute on a particular setting.
Experiment with browser page zoom to work with different parameter ranges!
The code is based on FM synthesis "by hand" from https://chuck.cs.princeton.edu/doc/examples/

global int mouseX, mouseY;
global Event playNote, finishNote;

SinOsc c => ADSR e => dac;
SinOsc m => blackhole;
440 => float cf;
880 => float mf => m.freq;
200 => float index;
// set A, D, S, and R
e.set( 10::ms, 8::ms, .5, 500::ms );
// (note: A, D, R are durations; S is a number between 0 and 1)

fun void runADSR() {
  while (true) {
    playNote => now;
    e.keyOn();
    finishNote => now;
    e.keyOff();
  }
}
spork ~ runADSR();
fun void runFM() {
  while( true ) {
    cf + (index * m.last()) => c.freq;
    1::samp => now;
  }
}
spork ~ runFM();
fun void RespondToMouseMovement() {
  while( true ) {
//    mouseX*0.5 => fm.freq;
    mouseY*0.5 => index;
    10::ms => now;
  }
}
spork ~ RespondToMouseMovement();
