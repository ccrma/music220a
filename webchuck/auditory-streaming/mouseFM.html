live code 
  
global int mouseX, mouseY;

FMFS fm; // instantiate an FMFS
fm.out => dac;
fm.car.gain(1.0);

// arguments are duration, pitch, pitch ratio, carrier envelope, modulator ratio, index, modulator envelope
[.01,.4,1.0,.1] @=> float carrierADSR[];
[.001,.2,1.0,.1] @=> float modulatorADSR[];
spork ~fm.playFM( 2000::second, 440.0, 1.0, carrierADSR, 1.0, 10.0, modulatorADSR);

// frequency is mapped to the x position of the mouse
fun void RespondToMouseMovement() {
  while( true ) {
//    1000 + 0.3 * mouseX => fm.freq;
    mouseX/100.0 => fm.index;
    mouseY/100.0 => fm.ratio;
//    <<< fm.index >>>; // 0 - 1126
    10::ms => now;
  }
}
spork ~ RespondToMouseMovement();
2000::second => now;
